    <script>
        var currentIndex = 0;

        var kanjiList = [
            { kanji: '果', meaning: 'Fruit', reading: 'か' },
            { kanji: '葡', meaning: 'Grape', reading: 'ぶ' },
            { kanji: '桃', meaning: 'Peach', reading: 'もも' },
            // Add more fruit kanji objects here
        ];

        var kanjiElement = document.getElementById("kanji");
        var meaningElement = document.getElementById("meaning");
        var readingElement = document.getElementById("reading");
        var showReadingButton = document.getElementById("show-reading");
        var prevButton = document.getElementById("prev-button");
        var nextButton = document.getElementById("next-button");
        var returnButton = document.getElementById("return-button");

        function showKanji() {
            var currentKanji = kanjiList[currentIndex];
            kanjiElement.innerHTML = currentKanji.kanji;
            meaningElement.innerHTML = currentKanji.meaning;
            readingElement.innerHTML = currentKanji.reading;
            readingElement.style.display = "none";
        }

        function showReading() {
            readingElement.style.display = "block";
        }

        function prevKanji() {
            currentIndex = (currentIndex - 1 + kanjiList.length) % kanjiList.length;
            showKanji();
        }

        function nextKanji() {
            currentIndex = (currentIndex + 1) % kanjiList.length;
            showKanji();
        }

        showKanji();

                showReadingButton.addEventListener("click", showReading);
        prevButton.addEventListener("click", prevKanji);
        nextButton.addEventListener("click", nextKanji);
        returnButton.addEventListener("click", function() {
            location.href = "index.html";
        });
    </script>
